# 合并表头

```tsx
import React from 'react'
import { AySearchTable, AyAction, AyCtrl, AySearchTableField } from 'amiya'
import { AyTableCtrlField } from 'amiya/lib/AyTable/ay-table'
import { listApi, addApi, updateApi, deleteApi, professionOptions } from '../api'
import 'antd/dist/antd.min.css'

const fields: Array<AySearchTableField> = [
  {
    title: '姓名',
    key: 'cname',
    search: {},
    dialog: {
      required: true
    }
  },
  {
    title: '英文名',
    key: 'name',
    search: {},
    dialog: {
      required: true,
      rules: [{ pattern: /^[a-z|A-Z|0-9]{1,}$/, message: '请输入字母或者数字' }]
    }
  },
  {
    title: '职业',
    key: 'profession',
    type: 'select',
    search: {},
    dialog: {},
    options: professionOptions
  },
  {
    title: '攻击力成长',
    key: 'phases', // 请给予这个地方 key，否则表头的自定义别名会出现意外结果
    table: {
      children: [
        {
          title: '精0',
          children: [
            {
              title: '等级1',
              align: 'center',
              render: (_, record) => {
                return record?.phases[0]?.attributesKeyFrames[0]?.data?.atk
              }
            },
            {
              title: '满级',
              align: 'center',
              render: (_, record) => {
                return record?.phases[0]?.attributesKeyFrames[1]?.data?.atk
              }
            }
          ]
        },
        {
          title: '精1',
          children: [
            {
              title: '等级1',
              align: 'center',
              render: (_, record) => {
                return record?.phases[1]?.attributesKeyFrames[0]?.data?.atk
              }
            },
            {
              title: '满级',
              align: 'center',
              render: (_, record) => {
                return record?.phases[1]?.attributesKeyFrames[1]?.data?.atk
              }
            }
          ]
        },
        {
          title: '精2',
          children: [
            {
              title: '等级1',
              align: 'center',
              render: (_, record) => {
                return record?.phases[2]?.attributesKeyFrames[0]?.data?.atk
              }
            },
            {
              title: '满级',
              align: 'center',
              render: (_, record) => {
                return record?.phases[2]?.attributesKeyFrames[1]?.data?.atk
              }
            }
          ]
        }
      ]
    }
  }
]

const CtrlField: AyTableCtrlField = {
  width: 200,
  render: (value, record) => {
    return (
      <AyCtrl>
        <AyAction record={record} action="update">
          编辑
        </AyAction>
        <AyAction record={record} action="delete">
          删除
        </AyAction>
        <AyAction record={record} action="view">
          详情
        </AyAction>
      </AyCtrl>
    )
  }
}

export default function Demo() {
  return (
    <AySearchTable
      title="表格标题"
      selectionType="checkbox"
      api={listApi}
      fields={fields}
      ctrl={CtrlField}
      deleteApi={deleteApi}
      dialogFormExtend={{
        fields: fields,
        updateApi,
        addApi
      }}
    >
      <AyAction action="batch-delete">批量删除</AyAction>
      <AyAction action="add">新建</AyAction>
    </AySearchTable>
  )
}
```
